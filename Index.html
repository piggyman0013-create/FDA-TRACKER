const SHEET_NAME = 'Cases';

function output(data) {
  return ContentService
    .createTextOutput(JSON.stringify(data))
    .setMimeType(ContentService.MimeType.JSON);
}

function doGet(e) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = ss.getSheetByName(SHEET_NAME);

  if (e.parameter && e.parameter.action) {
    const action = e.parameter.action;
    const p = JSON.parse(e.parameter.data || '{}');

    if (action === 'add') {
      sheet.appendRow([p.id,p.product,p.supplier,p.stage,p.submitDate,
        p.deadline,p.officer,p.skynetRef,p.note,p.deviceClass,p.pathway,
        p.etaDate,new Date().toISOString()]);
      return output({ success: true, action: 'added' });
    }

    if (action === 'update') {
      const data = sheet.getDataRange().getValues();
      for (let i = 1; i < data.length; i++) {
        if (String(data[i][0]) === String(p.id)) {
          sheet.getRange(i+1,2).setValue(p.product||'');
          sheet.getRange(i+1,3).setValue(p.supplier||'');
          sheet.getRange(i+1,4).setValue(p.stage||1);
          sheet.getRange(i+1,5).setValue(p.submitDate||'');
          sheet.getRange(i+1,6).setValue(p.deadline||'');
          sheet.getRange(i+1,7).setValue(p.officer||'');
          sheet.getRange(i+1,8).setValue(p.skynetRef||'');
          sheet.getRange(i+1,9).setValue(p.note||'');
          sheet.getRange(i+1,10).setValue(p.deviceClass||'');
          sheet.getRange(i+1,11).setValue(p.pathway||'');
          sheet.getRange(i+1,12).setValue(p.etaDate||'');
          sheet.getRange(i+1,13).setValue(new Date().toISOString());
          return output({ success: true, action: 'updated' });
        }
      }
      return output({ success: false, error: 'id not found' });
    }

    if (action === 'delete') {
      const data = sheet.getDataRange().getValues();
      for (let i = 1; i < data.length; i++) {
        if (String(data[i][0]) === String(p.id)) {
          sheet.deleteRow(i+1);
          return output({ success: true, action: 'deleted' });
        }
      }
      return output({ success: false, error: 'id not found' });
    }
  }

  const data = sheet.getDataRange().getValues();
  if (data.length < 2) return output({ success: true, data: [] });
  const headers = data[0];
  const rows = data.slice(1).filter(r => r[0]).map(row => {
    const obj = {};
    headers.forEach((h,i) => obj[h] = row[i]===null?'':String(row[i]));
    return obj;
  });
  return output({ success: true, data: rows });
}
